        const MAX_LENGTH = 50; 


        // 根據數據生成 HTML 卡片

        function renderQuotes(quotes) {

            container.innerHTML = ''; // 清空現有內容


            quotes.forEach((quote, index) => {

                const isLong = quote.content.length > MAX_LENGTH;

                let displayContent = quote.content;

                

                // 決定分類顏色 (Pastel Colors)

                let categoryColors = {};

                switch(quote.category) {

                    // 決策邏輯與創新思維: Pastel Blue

                    case 'decision': 

                        categoryColors = { bg: 'bg-blue-100', text: 'text-blue-700', border: 'border-blue-400', tab: 'border-blue-600' }; 

                        break;

                    // 心態轉變與認知升維: Pastel Green

                    case 'mindset': 

                        categoryColors = { bg: 'bg-green-100', text: 'text-green-700', border: 'border-green-400', tab: 'border-green-600' }; 

                        break;

                    // 未來競爭與領導變革: Pastel Purple

                    case 'future': 

                        categoryColors = { bg: 'bg-purple-100', text: 'text-purple-700', border: 'border-purple-400', tab: 'border-purple-600' }; 

                        break;

                    default: 

                        categoryColors = { bg: 'bg-gray-100', text: 'text-gray-700', border: 'border-gray-400', tab: 'border-gray-600' };

                }


                // 截斷內容並加上省略號

                if (isLong) {

                    displayContent = quote.content.substring(0, MAX_LENGTH) + '...';

                }


                // 決定分類文字

                let categoryText;

                switch(quote.category) {

                    case 'decision': categoryText = '決策邏輯與創新思維'; break;

                    case 'mindset': categoryText = '心態轉變與認知升維'; break;

                    case 'future': categoryText = '未來競爭與領導變革'; break;

                    default: categoryText = '未分類';

                }


                // 分割日期和時間

                const [datePart, timePart] = quote.date.split(' ');



                const cardHtml = `

                    <div class="quote-card ${categoryColors.bg} rounded-xl p-6 flex flex-col justify-between border-t-4 ${categoryColors.border} shadow-md" data-category="${quote.category}">

                        <div>

                            <span class="inline-block text-xs font-medium px-3 py-1 rounded-full bg-white ${categoryColors.text} ring-1 ring-${categoryColors.tab} mb-3">

                                ${categoryText}

                            </span>

                            <blockquote class="text-lg font-medium text-gray-800 italic leading-relaxed">

                                “${displayContent}”

                            </blockquote>

                        </div>

                        <!-- 底部區域：展開按鈕 (左) 與名字/日期 (右)，使用 flex items-center 實現垂直置中對齊 -->

                        <div class="mt-4 pt-3 border-t border-gray-200 flex items-center justify-between">

                            ${isLong ? `

                                <button class="text-sm ${categoryColors.text} hover:text-blue-800 font-semibold focus:outline-none transition duration-150 ease-in-out" onclick="event.stopPropagation(); openModal(${index})">

                                    [ 點選展開完整內容 ]

                                </button>

                            ` : '<div></div>'}

                            <div class="text-right">

                                <p class="text-sm text-gray-800 font-bold">

                                    - ${quote.name}

                                </p>

                                <p class="text-xs text-gray-500 mt-1">

                                    ${datePart || ''} ${timePart || ''}

                                </p>

                            </div>

                        </div>

                    </div>

                `;

                container.innerHTML += cardHtml;

            });

        }


        // 顯示指定類別的 Quote

        function showCategory(category) {

            const filteredQuotes = category === 'all'

                ? originalQuotes

                : originalQuotes.filter(q => q.category === category);


            renderQuotes(filteredQuotes);

            

            // 更新 Tab 樣式

            document.querySelectorAll('.category-tab').forEach(btn => {

                const btnCategory = btn.getAttribute('onclick').match(/'([^']+)'/)[1];

                let btnColorClass = 'text-blue-600 border-blue-600 hover:bg-blue-50'; // 預設藍色


                // 根據分類設定對應的顏色

                if (btnCategory === 'decision') btnColorClass = 'text-blue-600 border-blue-600 hover:bg-blue-50';

                else if (btnCategory === 'mindset') btnColorClass = 'text-green-600 border-green-600 hover:bg-green-50';

                else if (btnCategory === 'future') btnColorClass = 'text-purple-600 border-purple-600 hover:bg-purple-50';

                else if (btnCategory === 'all') btnColorClass = 'text-gray-600 border-gray-600 hover:bg-gray-50';


                // 樣式調整：與卡片色系一致 (border/text color)

                if (btnCategory === category) {

                    // Active state: 採用強調色 (Primary Blue)

                    btn.className = 'category-tab bg-blue-600 text-white px-4 py-2 rounded-full font-semibold text-sm transition duration-300 hover:bg-blue-700';

                } else {

                    // Inactive state: 採用分類色系的邊框和文字顏色

                    btn.className = `category-tab bg-white border ${btnColorClass} px-4 py-2 rounded-full font-semibold text-sm transition duration-300`;

                }

            });

        }


        // 開啟模態框

        function openModal(index) {

            const quote = originalQuotes[index];

            modalQuoteContent.textContent = quote.content;

            modalQuoteDate.textContent = `發言時間：${quote.date}`;

            modal.classList.remove('hidden');

            modal.classList.add('flex');

        }


        // 關閉模態框

        function closeModal() {

            modal.classList.add('hidden');

            modal.classList.remove('flex');

        }
